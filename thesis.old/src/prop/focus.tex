
\fboxsep=8pt

\newcommand{\PIdP}{
  \Infer[$Id$^+]
  {\PBFSeq{\Gamma,\Susp{A^+}}{\cdot}{\Foc{A^+}}}
  {}
}

\newcommand{\PIdN}{
  \Infer[$Id$^-]
  {\PBFSeq{\Gamma, \Foc{A^-}}{\cdot}{\Susp{A^-}}}
  {}
}

\newcommand{\PFocusL}{
  \Infer[$Focus-L$]
  {\PBFSeq{\Gamma,A^-}{\cdot}{\gamma}}
  {\PBFSeq{\Gamma, A^-, \Foc{A^-}}{\cdot}{\gamma}}
}

\newcommand{\PFocusR}{
  \Infer[$Focus-R$]
  {\PBFSeq{\Gamma}{\cdot}{A}}
  {\PBFSeq{\Gamma}{\cdot}{\Foc{A}}}
}

\newcommand{\PEtaRRule}{
  \Infer[\eta$-R$]
  {\PBFSeq{\Gamma}{\Omega}{p^-}}
  {\PBFSeq{\Gamma}{\Omega}{\Susp{p^-}}}
}

\newcommand{\PEtaLRule}{
  \Infer[\eta$-L$]
  {\PBFSeq{\Gamma}{p^+, \Omega}{\gamma}}
  {\PBFSeq{\Gamma, \Susp{p^+}}{\Omega}{\gamma}}
}

\newcommand{\PDownRRule}{
  \Infer[\Down$-R$]
  {\PBFSeq{\Gamma}{\cdot}{\Foc{\Down A^-}}}
  {\PBFSeq{\Gamma}{A^-}}
}

\newcommand{\PDownLRule}{
  \Infer[\Down-$L$]
  {\PBFSeq{\Gamma}{\Down A^-,\Omega}{\gamma}}
  {\PBFSeq{\Gamma, A^-}{\Omega}{\gamma}}
}

\newcommand{\PUpRRule}{
  \Infer[\Up$-R$]
  {\PBFSeq{\Gamma}{\Omega}{\Up A^+}}
  {\PBFSeq{\Gamma}{\Omega}{A^+}}
}

\newcommand{\PUpLRule}{
  \Infer[\Up$-L$]
  {\PBFSeq{\Gamma,\Foc{\Up A^+}}{\cdot}{\gamma}}
  {\PBFSeq{\Gamma}{A^+}{\gamma}}
}

\newcommand{\PAndNRRule}{
  \Infer[\AndN$-R$]
  {\PBFSeq{\Gamma}{\Omega}{A_1^- \AndN A_2^-}}
  {\PBFSeq{\Gamma}{\Omega}{A_1^-} & \PBFSeq{\Gamma}{\Omega}{A_2^-}}
}

\newcommand{\PAndNLaRule}{
  \Infer[\AndN$-L$_1]
  {\PBFSeq{\Gamma,  \Foc{A_1^- \AndN A_2^-}}{\cdot}{\gamma}}
  {\PBFSeq{\Gamma, \Foc{A_1^-}}{\cdot}{\gamma}}
}

\newcommand{\PAndNLbRule}{
  \Infer[\AndN$-L$_2]
  {\PBFSeq{\Gamma,  \Foc{A_1^- \AndN A_2^-}}{\cdot}{\gamma}}
  {\PBFSeq{\Gamma, \Foc{A_2^-}}{\cdot}{\gamma}}
}

\newcommand{\PTopNRRule}{
  \Infer[\TopN$-R$]
  {\PBFSeq{\Gamma}{\Omega}{\TopN}}
  {}
}

\newcommand{\PTopNLRule}{
  \mbox{No rule for $\TopN$-L}
}

\newcommand{\PAndPRRule}{
  \Infer[\AndP$-R$]
  {\PBFSeq{\Gamma}{\cdot}{\Foc{A_1^+ \AndP A_2^+}}}
  {\PBFSeq{\Gamma}{\cdot}{\Foc{A_1^+}} & \PBFSeq{\Gamma}{\cdot}{\Foc{A_2^+}}}
}

\newcommand{\PAndPLRule}{
  \Infer[\AndP$-L$]
  {\PBFSeq{\Gamma}{A_1^+ \AndP A_2^+, \Omega}{\gamma}}
  {\PBFSeq{\Gamma}{A_1^+, A_2^+, \Omega}{\gamma}}
}

\newcommand{\PTopPRRule}{
  \Infer[\TopP$-R$]
  {\PBFSeq{\Gamma}{\cdot}{\Foc{\TopP}}}
  {}
}

\newcommand{\PTopPLRule}{
  \Infer[\TopP-$L$]
  {\PBFSeq{\Gamma}{\TopP, \Omega}{\gamma}}
  {\PBFSeq{\Gamma}{\Omega}{\gamma}}
}

\newcommand{\PImpRRule}{
  \Infer[\Imp$-R$]
  {\PBFSeq{\Gamma}{\Omega}{A_1^+ \Imp A_2^-}}
  {\PBFSeq{\Gamma}{A_1^+, \Omega}{A_2^-}}
}

\newcommand{\PImpLRule}{
  \Infer[\Imp$-L$]
  {\PBFSeq{\Gamma,  \Foc{A_1^+ \Imp A_2^-}}{\cdot}{\gamma}}
  {\PBFSeq{\Gamma, A_1^+ \Imp A_2^-}{\cdot}{\Foc{A_1^+}} & \PBFSeq{\Gamma, \Foc{A_2^-}}{\cdot}{\gamma}}
}

\newcommand{\POrLRule}{
  \Infer[\Or$-L$]
  {\PBFSeq{\Gamma}{A_1^+ \Or A_2^+, \Omega}{\gamma}}
  {\PBFSeq{\Gamma}{A_1^+, \Omega}{\gamma} & \PBFSeq{\Gamma}{A_2^+, \Omega}{\gamma}}
}

\newcommand{\POrRaRule}{
  \Infer[\Or$-R$_1]
  {\PBFSeq{\Gamma}{\cdot}{\Foc{A_1^+ \Or A_2^+}}}
  {\PBFSeq{\Gamma}{\cdot}{\Foc{A_1^+}}}
}

\newcommand{\POrRbRule}{
  \Infer[\Or$-R$_2]
  {\PBFSeq{\Gamma}{\cdot}{\Foc{A_1^+ \Or A_2^+}}}
  {\PBFSeq{\Gamma}{\cdot}{\Foc{A_2^+}}}
}

\newcommand{\PBotRRule}{
  \mbox{No rule for $\Bot$-R}
}

\newcommand{\PBotLRule}{
  \Infer[$Init-R$]
  {\PBFSeq{\Gamma}{\Bot, \Omega}{\gamma}}
  {}
}
