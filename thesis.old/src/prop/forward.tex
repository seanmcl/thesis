
\newcommand{\PFInit}{
  \Infer[$Init$]{\FSeq{p}{p}}{}
}

\newcommand{\PFAndR}{
  \Infer[\And$-R$]
  {\FSeq{\cdot}{A_1 \And A_2}}
  {\FSeq{\cdot}{A_1} & \FSeq{\cdot}{A_2}}
}

\newcommand{\PFAndL}{
  \Infer[\And$-L$]
  {\FSeq{A_1 \And A_2}{\cdot}}
  {\FSeq{A_1, A_2}{\cdot}}
}

\newcommand{\PFImpR}{
  \Infer[\Imp$-R$]
  {\FSeq{\cdot}{A_1 \Imp A_2}}
  {\FSeq{A_1}{A_2}}
}

\newcommand{\PFImpL}{
  \Infer[\Imp$-L$]
  {\FSeq{A_1\Imp A_2}{\cdot}}
  {\FSeq{A_1\Imp A_2}{A_1} & \FSeq{A_2}{\cdot}}
}

\newcommand{\PFOrR}{
  \Infer[\Or$-R$_i]
  {\FSeq{\cdot}{A_1 \Or A_2}}
  {\FSeq{\cdot}{A_i}}
}

\newcommand{\PFOrL}{
  \Infer[\Or$-L$]
  {\FSeq{A_1 \Or A_2}{\cdot}}
  {\FSeq{A_1}{\cdot} & \FSeq{A_2}{\cdot}}
}

\newcommand{\PFTopR}{
  \Infer[\Top$-R$]
  {\FSeq{\cdot}{\Top}}
  {}
}

\newcommand{\PFBotL}{
  \Infer[\Bot$-L$]
  {\FSeq{\Bot}{\cdot}}
  {}
}
