
\newcommand{\PBInitRule}{
  \Infer[$Init$]
  {\NSeq{\Left{p},\Right{p}}}
  {}
}

\newcommand{\PBAndRRule}{
  \Infer[\And$-F$]
  {\NSeq{\Right{A_1 \And A_2}}}
  {\NSeq{\Right{A_1}} & \NSeq{\Right{A_2}}}
}

\newcommand{\PBAndLRule}{
  \Infer[\And$-T$]
  {\NSeq{\Left{A_1 \And A_2}}}
  {\NSeq{\Left{A_1}, \Left{A_2}}}
}

\newcommand{\PBImpRRule}{
  \Infer[\Imp$-F$]
  {\NSeq{\Right{A_1 \Imp A_2}}}
  {\NSeq{\Left{A_1},\Right{A_2}}}
}

\newcommand{\PBImpLRule}{
  \Infer[\Imp$-T$]
  {\NSeq{\Left{A_1\Imp A_2}}}
  {\NSeq{\Left{A_1\Imp A_2},\Right{A_1}} & \NSeq{\Left{A_2}}}
}

\newcommand{\PBOrRLRule}{
  \Infer[\Or$-F$_1]
  {\NSeq{\Right{A_1 \Or A_2}}}
  {\NSeq{\Right{A_1}}}
}

\newcommand{\PBOrRRRule}{
  \Infer[\Or$-F$_2]
  {\NSeq{\Right{A_1 \Or A_2}}}
  {\NSeq{\Right{A_2}}}
}

\newcommand{\PBOrLRule}{
  \Infer[\Or$-T$]
  {\NSeq{\Left{A_1 \Or A_2}}}
  {\NSeq{\Left{A_1}} & \NSeq{\Left{A_2}}}
}

\newcommand{\PBTopRRule}{
  \Infer[\Top$-F$]
  {\NSeq{\Right{\Top}}}
  {}
}

\newcommand{\PBBotLRule}{
  \Infer[\Bot$-T$]
  {\NSeq{\Left{\Bot}}}
  {}
}
