
\newcommand{\FBInit}{
  \Infer[$Init$]
  {\FSeq{\Gamma,p}{p}}
  {\Delta\Entails\Gamma,p~\Ok}
}

\newcommand{\FBAndR}{
  \Infer[\And$-R$]
  {\FSeq{\Gamma}{A_1 \And A_2}}
  {\FSeq{\Gamma}{A_1} & \FSeq{\Gamma}{A_2}}
}

\newcommand{\FBAndL}{
  \Infer[\And$-L$]
  {\FSeq{A_1 \And A_2}{\gamma}}
  {\FSeq{A_1, A_2}{\gamma}}
}

\newcommand{\FBImpR}{
  \Infer[\Imp$-R$]
  {\FSeq{\Gamma}{A_1 \Imp A_2}}
  {\FSeq{\Gamma,A_1}{A_2}}
}

\newcommand{\FBImpL}{
  \Infer[\Imp$-L$]
  {\FSeq{\Gamma,A_1\Imp A_2}{\gamma}}
  {\FSeq{\Gamma,A_1\Imp A_2}{A_1} & \FSeq{\Gamma,A_2}{\gamma}}
}

\newcommand{\FBOrRL}{
  \Infer[\Or$-R$_1]
  {\FSeq{\Gamma}{A_1 \Or A_2}}
  {\FSeq{\Gamma}{A_1}}
}

\newcommand{\FBOrRR}{
  \Infer[\Or$-R$_2]
  {\FSeq{\Gamma}{A_1 \Or A_2}}
  {\FSeq{\Gamma}{A_2}}
}

\newcommand{\FBOrL}{
  \Infer[\Or$-L$]
  {\FSeq{A_1 \Or A_2}{\gamma}}
  {\FSeq{A_1}{\gamma} & \FSeq{\Gamma, A_2}{\gamma}}
}

\newcommand{\FBTopR}{
  \Infer[\Top$-R$]
  {\FSeq{\Gamma}{\Top}}
  {\Delta\Entails\Gamma~\Ok}
}

\newcommand{\FBTopL}{\mbox{No rule for $\Top$-L}}

\newcommand{\FBBotL}{
  \Infer[\Bot$-L$]
  {\FSeq{\Gamma,\Bot}{\gamma}}
  {\Delta\Entails\Gamma,\gamma~\Ok}
}

\newcommand{\FBBotR}{\mbox{No rule for $\Bot$-R}}

\newcommand{\FBAllR}{
  \Infer[\All$-R$]
  {\FSeq{\Gamma}{\All x:\sigma.\ A(x)}}
  {\FSeq[\Delta,a:\sigma]{\Gamma}{A(a)}}
}

\newcommand{\FBAllL}{
  \Infer[\All$-L$]
  {\FSeq{\Gamma, \All x:\sigma.\ A(x)}{\gamma}}
  {\FSeq{\Gamma, A(t)}{\gamma} & \Delta\Entails t:\sigma}
}

\newcommand{\FBExL}{
  \Infer[\Ex$-L$^a]
  {\FSeq{\Gamma, \Ex x:\sigma.\ A(x)}{\gamma}}
  {\FSeq[\Delta,a:\sigma]{\Gamma, A(a)}{\gamma}}
}

\newcommand{\FBExR}{
  \Infer[\Ex$-R$]
  {\FSeq{\Gamma}{\Ex x:\sigma.\ A(x)}}
  {\FSeq{\Gamma}{A(t)} & \Delta\Entails t:\sigma}
}
