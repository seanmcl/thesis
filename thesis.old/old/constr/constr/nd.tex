
\newcommand{\CNAscribe}{
  \Infer[$Ascribe$]
  {\CNd{\Gamma}{M\ASet{A\Ueq A'} : A'}}
  {\CNd{\Gamma}{M : A} & \Gamma\Entails A\Ueq A'}
}

\newcommand{\CNDomain}{
  \Infer[$Domain$]
  {\CNd{\Gamma}{[E] : E}}
  {\Gamma\Entails E}
}
